# 확인 명령어

---

- 장치 확인

```bash
cat /etc/os-release
uname -a
uname -m

cat /etc/nv_tegra_release
dpkg-query --show nvidia-l4t-core
---

echo "===== OS ====="
cat /etc/os-release
uname -a
cat /proc/device-tree/model 2>/dev/null
cat /etc/nv_tegra_release 2>/dev/null

echo "===== Devices ====="
lsusb
lsusb -t
lspci 2>/dev/null
lsblk

echo "===== Dev nodes ====="
ls -l /dev/gpiochip* 2>/dev/null
ls -l /dev/i2c* 2>/dev/null
ls -l /dev/spidev* 2>/dev/null
ls -l /dev/ttyUSB* /dev/ttyACM* /dev/ttyTHS* 2>/dev/null
ls -l /dev/video* 2>/dev/null

echo "===== Network / CAN ====="
ip link show
ip -details link show | grep -A 20 -i can

echo "===== Kernel logs ====="
dmesg | grep -Ei "gpio|i2c|spi|tty|usb|can|mttcan|lidar|video|camera" | tail -n 100

```

- wifi 연결

```bash
nmcli device wifi list
nmcli device wifi connect <SSID> password <PASSWORD>
ifconfig
```
